def move
  puts "Move parameters are #{params}"
  rm=RouteMember.find(params['rm'])
  if !rm.nil? then
    rm.insert_at(params[:position].to_i)
    puts "update rm position to #{rm.position}"
    rm.save
  else
    puts "rm is nil for #{params}"
  end
  head :ok
end
